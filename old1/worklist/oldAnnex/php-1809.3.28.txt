//$test是单个日期 2008-8-8
function testDay($date){
	//切割成年月日
	$myday = explode("-",$date);
	//验证年月日是否完整，判定数组的大小 是否为3
	if(count($myday)!=3){
		echo "<font color='red'>".$date." 年月日不完整！</font><br/>";
		return;
	}
	//判定年月日是否为数字
	if(!is_numeric($myday[0])){
		echo "<font color='red'>".$date." 年不是数字！</font><br/>";
		return;
	}
	if(!is_numeric($myday[1])){
		echo "<font color='red'>".$date." 月不是数字！</font><br/>";
		return;
	}
	if(!is_numeric($myday[2])){
		echo "<font color='red'>".$date." 日不是数字！</font><br/>";
		return;
	}
	//判定年月日是否为整数
	if(!is_int(0+$myday[0])){
		echo "<font color='red'>".$date." 年不是整数！</font><br/>";
		return;
	}
	if(!is_int(0+$myday[1])){
		echo "<font color='red'>".$date." 月不是整数！</font><br/>";
		return;
	}
	if(!is_int(0+$myday[2])){
		echo "<font color='red'>".$date."日不是整数！</font><br/>";
		return;
	}
	//检测日期是否存在
	if(checkdate($myday[1], $myday[2], $myday[0])){
		echo $date."日期合法的！<br/>";
	}else{
		echo "<font color='red'>".$date."日期不存在！</font><br/>";
		return;
	}
	

}

//循环遍历所有数组元素
$data = array("2008-8-8","2008-08-08","2011-02-29","2008.88-8-8","2011-13-8","aaa-8-8","1999-8");
foreach ($data as $date){
	testDay($date);
}
